\begin{Pro}
   Convierte las siguientes expresiones regulares en autómatas finitos deterministas (DFA):
   \begin{enumerate}
       \item $[ab]^*$
       \item $(a?b^*)^*$
       \item $[ab]^*abb[ab]^*$
    \end{enumerate}
\end{Pro}
